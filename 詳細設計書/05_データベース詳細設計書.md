```sql
-- ============================================================
-- ST管理システム データベース初期化スクリプト
-- 対象: PostgreSQL 16
-- 文字コード: UTF-8
-- ============================================================

-- データベース作成 (手動実行)
-- CREATE DATABASE st_management ENCODING 'UTF8';

-- ============================================================
-- 1. user (ユーザー)
-- ============================================================
CREATE TABLE IF NOT EXISTS "user" (
    id          BIGSERIAL       PRIMARY KEY,
    username    VARCHAR(50)     NOT NULL,
    password    VARCHAR(255)    NOT NULL,
    role        VARCHAR(20)     NOT NULL DEFAULT 'USER',
    create_time TIMESTAMP       NOT NULL DEFAULT NOW(),
    update_time TIMESTAMP       NOT NULL DEFAULT NOW(),

    CONSTRAINT uk_user_username UNIQUE (username),
    CONSTRAINT chk_user_role CHECK (role IN ('ADMIN', 'USER'))
);

COMMENT ON TABLE  "user"             IS 'ログインユーザー';
COMMENT ON COLUMN "user".id          IS '主キー';
COMMENT ON COLUMN "user".username    IS 'ログインID（一意）';
COMMENT ON COLUMN "user".password    IS 'パスワード（BCryptハッシュ）';
COMMENT ON COLUMN "user".role        IS 'ロール: ADMIN/USER';
COMMENT ON COLUMN "user".create_time IS '作成日時';
COMMENT ON COLUMN "user".update_time IS '更新日時';

-- ============================================================
-- 2. employee (社員)
-- ============================================================
CREATE TABLE IF NOT EXISTS employee (

COMMENT ON TABLE  employee                    IS '社員情報';
COMMENT ON COLUMN employee.id                 IS '主キー';

COMMENT ON COLUMN bank_account.account_holder IS '口座名義人';
COMMENT ON COLUMN bank_account.create_time   IS '作成日時';
COMMENT ON COLUMN bank_account.update_time   IS '更新日時';

-- ============================================================
-- 4. attendance (勤務)
-- ============================================================
CREATE TABLE IF NOT EXISTS attendance (
    id              BIGSERIAL       PRIMARY KEY,
    employee_id     BIGINT          NOT NULL,
    work_date       DATE            NOT NULL,
    work_hours      DECIMAL(4,1)    NOT NULL DEFAULT 0.0,
    overtime_hours  DECIMAL(4,1)    NOT NULL DEFAULT 0.0,
    status          VARCHAR(20)     NOT NULL,
    remark          VARCHAR(500),
    create_time     TIMESTAMP       NOT NULL DEFAULT NOW(),
    update_time     TIMESTAMP       NOT NULL DEFAULT NOW(),

    CONSTRAINT fk_attendance_employee
        FOREIGN KEY (employee_id) REFERENCES employee(id),

    CONSTRAINT chk_attendance_status
        CHECK (status IN ('出勤', '欠勤', '休暇')),
    CONSTRAINT chk_work_hours
        CHECK (work_hours >= 0 AND work_hours <= 24),
    CONSTRAINT chk_overtime_hours
        CHECK (overtime_hours >= 0 AND overtime_hours <= 12),
    CONSTRAINT uk_attendance_emp_date
        UNIQUE (employee_id, work_date)
);

COMMENT ON TABLE  attendance               IS '勤務記録';
COMMENT ON COLUMN attendance.id            IS '主キー';
COMMENT ON COLUMN attendance.employee_id   IS '社員ID（FK → employee.id）';
COMMENT ON COLUMN attendance.work_date     IS '勤務日付';
COMMENT ON COLUMN attendance.work_hours    IS '所定勤務時間（h）';
COMMENT ON COLUMN attendance.overtime_hours IS '残業時間（h）';
COMMENT ON COLUMN attendance.status        IS '勤務状態: 出勤/欠勤/休暇';
COMMENT ON COLUMN attendance.remark        IS '備考・特記事項';
COMMENT ON COLUMN attendance.create_time   IS '作成日時';
COMMENT ON COLUMN attendance.update_time   IS '更新日時';

-- ============================================================
-- 5. invoice (請求書)
-- ============================================================
CREATE TABLE IF NOT EXISTS invoice (
    id              BIGSERIAL       PRIMARY KEY,
    invoice_number  VARCHAR(50)     NOT NULL,
    customer_id     BIGINT          NOT NULL,
    year            INT             NOT NULL,
    month           INT             NOT NULL,
    amount          DECIMAL(12,2)   NOT NULL DEFAULT 0.00,
    status          VARCHAR(20)     NOT NULL,
    remark          TEXT,
    create_time     TIMESTAMP       NOT NULL DEFAULT NOW(),
    update_time     TIMESTAMP       NOT NULL DEFAULT NOW(),

    CONSTRAINT fk_invoice_customer
        FOREIGN KEY (customer_id) REFERENCES customer(id),
    CONSTRAINT uk_invoice_number
        UNIQUE (invoice_number),
    CONSTRAINT chk_invoice_month
        CHECK (month BETWEEN 1 AND 12),
    CONSTRAINT chk_invoice_amount
        CHECK (amount >= 0),
    CONSTRAINT chk_invoice_status
        CHECK (status IN ('下書き', '送付済', '入金済'))
);

COMMENT ON TABLE  invoice                IS '請求書';
COMMENT ON COLUMN invoice.id             IS '主キー';
COMMENT ON COLUMN invoice.invoice_number IS '請求書識別番号（一意）';
COMMENT ON COLUMN invoice.customer_id    IS '顧客ID（FK → customer.id）';
COMMENT ON COLUMN invoice.year           IS '請求対象年';
COMMENT ON COLUMN invoice.month          IS '請求対象月（1-12）';
COMMENT ON COLUMN invoice.amount         IS '請求金額（円）';
COMMENT ON COLUMN invoice.status         IS 'ステータス: 下書き/送付済/入金済';
COMMENT ON COLUMN invoice.remark         IS '備考・摘要';
COMMENT ON COLUMN invoice.create_time    IS '作成日時';
COMMENT ON COLUMN invoice.update_time    IS '更新日時';

-- ============================================================
-- 6. order_documents (注文書)
-- ============================================================
CREATE TABLE IF NOT EXISTS order_documents (
    id          BIGSERIAL       PRIMARY KEY,
    invoice_id  BIGINT          NOT NULL,
    file_name   VARCHAR(255)    NOT NULL,
    file_path   VARCHAR(500)    NOT NULL,
    file_size   BIGINT          NOT NULL DEFAULT 0,
    create_time TIMESTAMP       NOT NULL DEFAULT NOW(),
    update_time TIMESTAMP       NOT NULL DEFAULT NOW(),

    CONSTRAINT fk_ordoc_invoice
        FOREIGN KEY (invoice_id) REFERENCES invoice(id)
        ON DELETE CASCADE
);

COMMENT ON TABLE  order_documents             IS '注文書（添付資料）';
COMMENT ON COLUMN order_documents.id          IS '主キー';
COMMENT ON COLUMN order_documents.invoice_id  IS '請求書ID（FK → invoice.id, CASCADE）';
COMMENT ON COLUMN order_documents.file_name   IS 'オリジナルファイル名';
COMMENT ON COLUMN order_documents.file_path   IS 'サーバー上の保存パス';
COMMENT ON COLUMN order_documents.file_size   IS 'ファイルサイズ（byte）';
COMMENT ON COLUMN order_documents.create_time IS '作成日時';
COMMENT ON COLUMN order_documents.update_time IS '更新日時';
```

## 2. インデックス設計

```sql
-- ============================================================
-- インデックス作成
-- ============================================================

-- user テーブル
CREATE UNIQUE INDEX idx_user_username ON "user" (username);

-- employee テーブル
-- （UNIQUE制約により idx_employee_code は自動生成）
CREATE INDEX idx_employee_name       ON employee (name);
CREATE INDEX idx_employee_department ON employee (department);

-- customer テーブル
-- （UNIQUE制約により idx_customer_code は自動生成）
CREATE INDEX idx_customer_company ON customer (company_name);

-- attendance テーブル
CREATE INDEX idx_att_employee_date ON attendance (employee_id, work_date);
CREATE INDEX idx_att_date          ON attendance (work_date);

-- invoice テーブル
-- （UNIQUE制約により idx_invoice_number は自動生成）
CREATE INDEX idx_invoice_customer_ym ON invoice (customer_id, year, month);

-- order_documents テーブル
CREATE INDEX idx_ordoc_invoice ON order_documents (invoice_id);
```

## 4. Entity クラス設計 (JPA)

### 4.1 Employee Entity

```java
@Entity
@Table(name = "employee")
public class Employee {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Column(name = "employee_code", length = 20, nullable = false, unique = true)
    private String employeeCode;

    @Column(length = 100, nullable = false)
    private String name;

    @Column(length = 255)
    private String email;

    @Column(length = 20)
    private String phone;

    @Column(length = 500)
    private String address;

    @Column(length = 100)
    private String department;

    @Column(length = 100)
    private String position;

    // 銀行口座情報
    @Column(name = "bank_name", length = 200)
    private String bankName;

    @Column(name = "bank_branch", length = 200)
    private String bankBranch;

    @Column(name = "bank_account_type", length = 20)
    private String bankAccountType;

    @Column(name = "bank_account_number", length = 50)
    private String bankAccountNumber;

    @Column(name = "bank_account_holder", length = 100)
    private String bankAccountHolder;

    @Column(name = "attachment_path", length = 500)
    private String attachmentPath;

    @Column(name = "create_time", nullable = false)
    private LocalDateTime createTime;

    @Column(name = "update_time", nullable = false)
    private LocalDateTime updateTime;

    // @PrePersist / @PreUpdate で createTime/updateTime 自動設定
    @PrePersist
    protected void onCreate() {
        createTime = LocalDateTime.now();
        updateTime = LocalDateTime.now();
    }

    @PreUpdate
    protected void onUpdate() {
        updateTime = LocalDateTime.now();
    }

    // getters / setters ...
}
```

```java
public interface EmployeeRepository extends JpaRepository<Employee, Long> {

    Optional<Employee> findByEmployeeCode(String employeeCode);
```

## 7. トランザクション設計

### 7.1 トランザクション境界

| 処理 | トランザクション | 理由 |
|------|:--------------:|------|
| 一覧取得 (GET) | なし (readOnly) | データ不整合のリスクなし |
| 新規登録 (POST) | @Transactional | 単一テーブル挿入の原子性保証 |
| 更新 (PUT) | @Transactional | 存在チェック + 更新の原子性保証 |
| 削除 (DELETE) | @Transactional | レコード削除 + ファイル削除 + 関連レコード削除の原子性保証 |
| ファイルUP/DL | @Transactional | DB更新 + ファイルI/O の原子性保証 |

### 7.2 トランザクション分離レベル

デフォルト: `READ COMMITTED` (PostgreSQL デフォルト、Spring @Transactional のデフォルト)

| シナリオ | 分離レベル | 理由 |
|---------|:---------:|------|
| 通常CRUD | READ COMMITTED | PostgreSQLデフォルトで十分 |
| 請求書番号採番 | READ COMMITTED | 一意制約により重複防止 |
